The loan portfolio is synthetically generated in Python (numpy, pandas) with default probability driven primarily by credit score, then re-analyzed with pandas to validate the SQL findings. Full files are in the GitHub repo.

# credit score: normal distribution centered at 720, range 300-900
credit_score = int(np.clip(np.random.normal(720, 85), 300, 900))

# default probability driven primarily by credit score, with loan-type
# and region modifiers layered on top
score_risk = max(0.01, (750 - credit_score) / 1000)
type_modifier = {"Personal Loan": 1.3, "Auto Loan": 0.9, "Home Loan": 0.5,
                  "Business Loan": 1.4, "Gold Loan": 0.6, "Consumer Durable": 1.5}[ltype]
region_modifier = {"North": 1.0, "South": 0.85, "East": 1.15, "West": 0.9, "Central": 1.1}[region]
default_prob = min(0.55, score_risk * type_modifier * region_modifier)

roll = random.random()
if roll < default_prob * 0.4:
    dpd_bucket = "90+ DPD (NPA)"
    collections_amount = outstanding_principal * np.random.uniform(0.05, 0.25)
elif roll < default_prob * 0.6:
    dpd_bucket = "61-90 DPD"
    collections_amount = outstanding_principal * np.random.uniform(0.2, 0.5)
elif roll < default_prob:
    dpd_bucket = "1-30 DPD"
    collections_amount = outstanding_principal * np.random.uniform(0.6, 0.9)
else:
    dpd_bucket = "Current"
    collections_amount = outstanding_principal * np.random.uniform(0.9, 1.0)

Pandas analysis & risk segmentation

analysis.py

analysis.py 
def band(score):
    if score >= 800: return "800+"
    if score >= 750: return "750-799"
    if score >= 700: return "700-749"
    if score >= 650: return "650-699"
    if score >= 600: return "600-649"
    return "Below 600"

df["score_band"] = df["credit_score"].apply(band)
band_order = ["800+", "750-799", "700-749", "650-699", "600-649", "Below 600"]
score_stats = df.groupby("score_band").agg(
    total=("loan_id", "count"),
    npa=("dpd_bucket", lambda x: (x == "90+ DPD (NPA)").sum())
).reindex(band_order)
score_stats["npa_rate"] = 100 * score_stats["npa"] / score_stats["total"]

print(score_stats)
# Below 600 defaults at 8.60% vs 0.37% for 800+ -- a ~23x risk multiple

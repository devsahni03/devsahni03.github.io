rows = []
for i in range(N):
    dept = np.random.choice(departments, p=dept_weights)
    payer = np.random.choice(payers, p=payer_weights)
    billed = max(150, np.random.normal(dept_base[dept], dept_base[dept] * 0.35))

    # denial probability varies realistically by payer
    denial_prob = {
        "Aetna": 0.09, "UnitedHealthcare": 0.10, "Cigna": 0.085, "Medicare": 0.11,
        "Medicaid": 0.14, "BlueCross BlueShield": 0.095, "Self-Pay": 0.05
    }[payer]

    roll = random.random()
    if roll < denial_prob:
        claim_status = "Denied"
        allowed, paid = 0, 0
    elif roll < denial_prob + 0.06:
        claim_status = "Partially Paid"
        allowed = billed * np.random.uniform(0.4, 0.75)
        paid = allowed * np.random.uniform(0.6, 0.95)
    else:
        claim_status = "Paid"
        allowed = billed * np.random.uniform(0.82, 0.98)
        paid = allowed * np.random.uniform(0.95, 1.0)

    rows.append({
        "encounter_id": f"ENC{100000+i}", "department": dept, "payer": payer,
        "billed_amount": round(billed, 2),
        "paid_amount": round(paid, 2) if paid is not None else None,
        "claim_status": claim_status,
    })

df = pd.DataFrame(rows)
df.to_csv("data/healthcare_encounters.csv", index=False)

analysis.py 
df = pd.read_sql_query("SELECT * FROM encounters", conn)

total_billed = df["billed_amount"].sum() / 1e7
total_paid = df["paid_amount"].fillna(0).sum() / 1e7
leakage_pct = 100 * (total_billed - total_paid) / total_billed
denial_rate = 100 * (df["claim_status"] == "Denied").mean()

payer_stats = df.groupby("payer").agg(
    total=("encounter_id", "count"),
    denied=("claim_status", lambda x: (x == "Denied").sum())
)
payer_stats["denial_rate"] = 100 * payer_stats["denied"] / payer_stats["total"]
payer_stats = payer_stats.sort_values("denial_rate", ascending=False)

print(f"Revenue Leakage: {leakage_pct:.1f}%")
print(f"Overall Denial Rate: {denial_rate:.1f}%")
print(f"Highest-Denial Payer: {payer_stats['denial_rate'].idxmax()}")

# Export KPI data as JSON for the dashboard
import json
kpi = {

    "denial_rate": round(denial_rate, 1),
    "payer_denial": payer_stats["denial_rate"].round(1).to_dict(),
}
with open("dashboard/assets/kpi_data.json", "w") as f:
    json.dump(kpi, f, indent=2)
